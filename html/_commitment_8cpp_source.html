62 CommitmentProofOfKnowledge::CommitmentProofOfKnowledge(const IntegerGroupParams* aParams,
63  const IntegerGroupParams* bParams, const Commitment& a, const Commitment& b):
64  ap(aParams),bp(bParams)
65 {
66  Bignum r1, r2, r3;
67 
68  // First: make sure that the two commitments have the
69  // same contents.
70  if (a.getContents() != b.getContents()) {
71  throw ZerocoinException("Both commitments must contain the same value");
72  }
73 
74  // Select three random values "r1, r2, r3" in the range 0 to (2^l)-1 where l is:
75  // length of challenge value + max(modulus 1, modulus 2, order 1, order 2) + margin.
76  // We set "margin" to be a relatively generous security parameter.
77  //
78  // We choose these large values to ensure statistical zero knowledge.
79  uint32_t randomSize = COMMITMENT_EQUALITY_CHALLENGE_SIZE + COMMITMENT_EQUALITY_SECMARGIN +
80  std::max(std::max(this->ap->modulus.bitSize(), this->bp->modulus.bitSize()),
81  std::max(this->ap->groupOrder.bitSize(), this->bp->groupOrder.bitSize()));
82  Bignum maxRange = (Bignum(2).pow(randomSize) - Bignum(1));
83 
84  r1 = Bignum::randBignum(maxRange);
85  r2 = Bignum::randBignum(maxRange);
86  r3 = Bignum::randBignum(maxRange);
87 
88  // Generate two random, ephemeral commitments "T1, T2"
89  // of the form:
90  // T1 = g1^r1 * h1^r2 mod p1
91  // T2 = g2^r1 * h2^r3 mod p2
92  //
93  // Where (g1, h1, p1) are from "aParams" and (g2, h2, p2) are from "bParams".
94  Bignum T1 = this->ap->g().pow_mod(r1, this->ap->modulus).mul_mod((this->ap->h().pow_mod(r2, this->ap->modulus)), this->ap->modulus);
95  Bignum T2 = this->bp->g().pow_mod(r1, this->bp->modulus).mul_mod((this->bp->h().pow_mod(r3, this->bp->modulus)), this->bp->modulus);
96 
97  // Now hash commitment "A" with commitment "B" as well as the
98  // parameters and the two ephemeral commitments "T1, T2" we just generated
99  this->challenge = calculateChallenge(a.getCommitmentValue(), b.getCommitmentValue(), T1, T2);
100 
101  // Let "m" be the contents of the commitments "A, B". We have:
102  // A = g1^m * h1^x mod p1
103  // B = g2^m * h2^y mod p2
104  // T1 = g1^r1 * h1^r2 mod p1
105  // T2 = g2^r1 * h2^r3 mod p2
106  //
107  // Now compute:
108  // S1 = r1 + (m * challenge) -- note, not modular arithmetic
109  // S2 = r2 + (x * challenge) -- note, not modular arithmetic
110  // S3 = r3 + (y * challenge) -- note, not modular arithmetic
111  this->S1 = r1 + (a.getContents() * this->challenge);
112  this->S2 = r2 + (a.getRandomness() * this->challenge);
113  this->S3 = r3 + (b.getRandomness() * this->challenge);
114 
115  // We're done. The proof is S1, S2, S3 and "challenge", all of which
116  // are stored in member variables.
117 }

119 bool CommitmentProofOfKnowledge::Verify(const Bignum& A, const Bignum& B) const
120 {
121  // Compute the maximum range of S1, S2, S3 and verify that the given values are
122  // in a correct range. This might be an unnecessary check.
123  uint32_t maxSize = 64 * (COMMITMENT_EQUALITY_CHALLENGE_SIZE + COMMITMENT_EQUALITY_SECMARGIN +
124  std::max(std::max(this->ap->modulus.bitSize(), this->bp->modulus.bitSize()),
125  std::max(this->ap->groupOrder.bitSize(), this->bp->groupOrder.bitSize())));
126 
127  if ((uint32_t)this->S1.bitSize() > maxSize ||
128  (uint32_t)this->S2.bitSize() > maxSize ||
129  (uint32_t)this->S3.bitSize() > maxSize ||
130  this->S1 < Bignum(0) ||
131  this->S2 < Bignum(0) ||
132  this->S3 < Bignum(0) ||
133  this->challenge < Bignum(0) ||
134  this->challenge > (Bignum(2).pow(COMMITMENT_EQUALITY_CHALLENGE_SIZE) - Bignum(1))) {
135  // Invalid inputs. Reject.
136  return false;
137  }
138 
139  // Compute T1 = g1^S1 * h1^S2 * inverse(A^{challenge}) mod p1
140  Bignum T1 = A.pow_mod(this->challenge, ap->modulus).inverse(ap->modulus).mul_mod(
141  (ap->g().pow_mod(S1, ap->modulus).mul_mod(ap->h().pow_mod(S2, ap->modulus), ap->modulus)),
142  ap->modulus);
143 
144  // Compute T2 = g2^S1 * h2^S3 * inverse(B^{challenge}) mod p2
145  Bignum T2 = B.pow_mod(this->challenge, bp->modulus).inverse(bp->modulus).mul_mod(
146  (bp->g().pow_mod(S1, bp->modulus).mul_mod(bp->h().pow_mod(S3, bp->modulus), bp->modulus)),
147  bp->modulus);
148 
149  // Hash T1 and T2 along with all of the public parameters
150  Bignum computedChallenge = calculateChallenge(A, B, T1, T2);
151 
152  // Return success if the computed challenge matches the incoming challenge
153  if(computedChallenge == this->challenge) {
154  return true;
155  }
156 
157  // Otherwise return failure
158  return false;
159 }

161 const Bignum CommitmentProofOfKnowledge::calculateChallenge(const Bignum& a, const Bignum& b, const Bignum &commitOne, const Bignum &commitTwo) const {
162  CHashWriter hasher(0,0);
163 
164  // Hash together the following elements:
165  // * A string identifying the proof
166  // * Commitment A
167  // * Commitment B
168  // * Ephemeral commitment T1
169  // * Ephemeral commitment T2
170  // * A serialized instance of the commitment A parameters
171  // * A serialized instance of the commitment B parameters
172 
173  hasher << std::string(ZEROCOIN_COMMITMENT_EQUALITY_PROOF);
174  hasher << commitOne;
175  hasher << std::string("||");
176  hasher << commitTwo;
177  hasher << std::string("||");
178  hasher << a;
179  hasher << std::string("||");
180  hasher << b;
181  hasher << std::string("||");
182  hasher << *(this->ap);
183  hasher << std::string("||");
184  hasher << *(this->bp);
185 
186  // Convert the SHA256 result into a Bignum
187  // Note that if we ever change the size of the hash function we will have
188  // to update COMMITMENT_EQUALITY_CHALLENGE_SIZE appropriately!
189  return Bignum(hasher.GetHash());
190 }
