19  Accumulator& a, const AccumulatorWitness& witness, const SpendMetaData& m):
20  params(p),
21  denomination(coin.getPublicCoin().getDenomination()),
22  coinSerialNumber((coin.getSerialNumber())),
23  accumulatorPoK(&p->accumulatorParams),
24  serialNumberSoK(p),
25  commitmentPoK(&p->serialNumberSoKCommitmentGroup, &p->accumulatorParams.accumulatorPoKCommitmentGroup) {
26 
27  // Sanity check: let's verify that the Witness is valid with respect to
28  // the coin and Accumulator provided.
29  if (!(witness.VerifyWitness(a, coin.getPublicCoin()))) {
30  throw ZerocoinException("Accumulator witness does not verify");
31  }
32 
33  // 1: Generate two separate commitments to the public coin (C), each under
34  // a different set of public parameters. We do this because the RSA accumulator
35  // has specific requirements for the commitment parameters that are not
36  // compatible with the group we use for the serial number proof.
37  // Specifically, our serial number proof requires the order of the commitment group
38  // to be the same as the modulus of the upper group. The Accumulator proof requires a
39  // group with a significantly larger order.
40  const Commitment fullCommitmentToCoinUnderSerialParams(&p->serialNumberSoKCommitmentGroup, coin.getPublicCoin().getValue());
41  this->serialCommitmentToCoinValue = fullCommitmentToCoinUnderSerialParams.getCommitmentValue();
42 
43  const Commitment fullCommitmentToCoinUnderAccParams(&p->accumulatorParams.accumulatorPoKCommitmentGroup, coin.getPublicCoin().getValue());
44  this->accCommitmentToCoinValue = fullCommitmentToCoinUnderAccParams.getCommitmentValue();
45 
46  // 2. Generate a ZK proof that the two commitments contain the same public coin.
47  this->commitmentPoK = CommitmentProofOfKnowledge(&p->serialNumberSoKCommitmentGroup, &p->accumulatorParams.accumulatorPoKCommitmentGroup, fullCommitmentToCoinUnderSerialParams, fullCommitmentToCoinUnderAccParams);
48  cout << "GNOSIS DEBUG: commitmentPoK is " << this->commitmentPoK.GetSerializeSize(SER_NETWORK, PROTOCOL_VERSION) << " bytes" << endl;;
49 
50  // Now generate the two core ZK proofs:
51  // 3. Proves that the committed public coin is in the Accumulator (PoK of "witness")
52  this->accumulatorPoK = AccumulatorProofOfKnowledge(&p->accumulatorParams, fullCommitmentToCoinUnderAccParams, witness, a);
53  cout << "GNOSIS DEBUG: accPoK is " << this->accumulatorPoK.GetSerializeSize(SER_NETWORK, PROTOCOL_VERSION) << " bytes" << endl;;
54 
55  // 4. Proves that the coin is correct w.r.t. serial number and hidden coin secret
56  // (This proof is bound to the coin 'metadata', i.e., transaction hash)
57  this->serialNumberSoK = SerialNumberSignatureOfKnowledge(p, coin, fullCommitmentToCoinUnderSerialParams, signatureHash(m));
58  cout << "GNOSIS DEBUG: snSoK is " << this->serialNumberSoK.GetSerializeSize(SER_NETWORK, PROTOCOL_VERSION) << " bytes" << endl;;
59 }

73  // Verify both of the sub-proofs using the given meta-data
74  struct timeval tv0, tv1;
75  double elapsed;
76 
77  gettimeofday(&tv0, NULL);
78 
79  bool result_cPoK = commitmentPoK.Verify(serialCommitmentToCoinValue, accCommitmentToCoinValue);
80  gettimeofday(&tv1, NULL);

84 
85  bool result_accPoK = accumulatorPoK.Verify(a, accCommitmentToCoinValue);

89  (tv1.tv_usec - tv0.tv_usec) / 1e6;
90  cout << "GNOSIS DEBUG: accPoK time: " << elapsed << endl;
91 
92  bool result_snSoK = serialNumberSoK.Verify(coinSerialNumber, serialCommitmentToCoinValue, signatureHash(m));
93  tv0 = tv1;
94  gettimeofday(&tv1, NULL);
95  elapsed = (tv1.tv_sec - tv0.tv_sec) +
96  (tv1.tv_usec - tv0.tv_usec) / 1e6;
97  cout << "GNOSIS DEBUG: snSoK time: " << elapsed << endl;
98 
99  return (a.getDenomination() == this->denomination)

101  && result_accPoK
