119  // Generate a random serial number in the range 0...{q-1} where
120  // "q" is the order of the commitment group.
121  Bignum s = Bignum::randBignum(this->params->coinCommitmentGroup.groupOrder);
122 
123  // Generate a random number "r" in the range 0...{q-1}
124  Bignum r = Bignum::randBignum(this->params->coinCommitmentGroup.groupOrder);
125 
126  Bignum cc_g, cc_h; // generators g & h for coin commitment group
127  deriveGeneratorsFromSerialNumber(s, this->params->coinCommitmentGroup.modulus, this->params->coinCommitmentGroup.groupOrder, cc_g, cc_h);
128 
129  // Manually compute a Pedersen commitment to the serial number "s" under randomness "r"
130  // C = g^s * h^r mod p
131  Bignum commitmentValue = cc_g.pow_mod(s, this->params->coinCommitmentGroup.modulus).mul_mod(cc_h.pow_mod(r, this->params->coinCommitmentGroup.modulus), this->params->coinCommitmentGroup.modulus);
132 
133  // Repeat this process up to MAX_COINMINT_ATTEMPTS times until
134  // we obtain a prime number
135  for (uint32_t attempt = 0; attempt < MAX_COINMINT_ATTEMPTS; attempt++) {
136  // First verify that the commitment is a prime number
137  // in the appropriate range. If not, we'll throw this coin
138  // away and generate a new one.
139  if (commitmentValue.isPrime(ZEROCOIN_MINT_PRIME_PARAM) &&
140  commitmentValue >= params->accumulatorParams.minCoinValue &&
141  commitmentValue <= params->accumulatorParams.maxCoinValue) {
142  // Found a valid coin. Store it.
143  this->serialNumber = s;
144  this->randomness = r;

146

152  Bignum r_delta = Bignum::randBignum(this->params->coinCommitmentGroup.groupOrder);
153

155  // r = r + r_delta mod q
156  // C = C * h mod p
157  r = (r + r_delta) % this->params->coinCommitmentGroup.groupOrder;
158  commitmentValue = commitmentValue.mul_mod(cc_h.pow_mod(r_delta, this->params->coinCommitmentGroup.modulus), this->params->coinCommitmentGroup.modulus);
